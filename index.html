My solution to «Convert string to camel case» kata on Codewars: 

function toCamelCase(str) {
    let words = str.split(/[_-]/);
    let output = "";        
    for (i = 0; i < words.length; i++) {
        let word = words[i];
        if (i == 0) {
            output += word; // leave 1st word unchanged
        } else {
            output += word.charAt(0).toUpperCase()
                + word.slice(1);
        }
    }        
    return output;
}
